    }

    const calculateTimeLeft = () => {
        const now = new Date();
        const diff = deadline.getTime() - now.getTime();

            setTimeLeft({
                hours: 0,
                minutes: 0,
                seconds: 0,
                isClose: false,
                isPast: true
            });
            return;
        }
        
        const hours = Math.floor(diff / (1000 * 60 * 60));
        const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
        const seconds = Math.floor((diff % (1000 * 60)) / 1000);
        const isClose = hours < 1;
        
        setTimeLeft(prev => {
            // Only update if values changed significantly (more than 1 second difference)
            if (!prev || prev.hours !== hours || prev.minutes !== minutes || prev.seconds !== seconds) {
                return { hours, minutes, seconds, isClose, isPast: false };
            }
            return prev;
        });
    };

    calculateTimeLeft();
    const interval = setInterval(calculateTimeLeft, 1000);
    
    return () => clearInterval(interval);
