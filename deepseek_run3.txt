We are building a "Magic Paste" feature that parses tournament details from pasted text (from an email or website) and a "Review" page that shows the parsed details with an option to customize divisions.

Goals:
1. Parse pasted text client-side with high accuracy and instant feedback.

```typescript
function parseTournamentDetails(text: string): ParsedTournament {
  const result: ParsedTournament = {
    sanctionId: null,
    name: null,
    date: null,
    location: null,
    organizer: null,
    tournamentType: null,
  };

  const lines = text.split('\n').map(line => line.trim()).filter(line => line.length > 0);

  // First, look for the Sanction ID in the entire text (not just per line) because it might be split across lines.
  const sanctionIdPattern = /([A-Z]{2}-[A-Z]{2}-\d{6})|([A-Z]{2}[A-Z]{2}\d{6})/;
  const sanctionIdMatch = text.match(sanctionIdPattern);
  if (sanctionIdMatch) {
    result.sanctionId = sanctionIdMatch[1] || sanctionIdMatch[2];
  }

  // Now, for each line, try to extract other details.
  for (const line of lines) {
    // If we haven't found the tournament name, try to extract it from the line that doesn't match other patterns.
    if (!result.name) {
      // Check if the line looks like a tournament name (doesn't contain typical labels and is not too long).
      if (!line.match(/sanction|id|date|location|organizer|type|format|rounds|time/i) && line.length < 100) {
        result.name = line;
      }
    }

    // Date pattern: look for a date in the line.
    const datePattern = /(?:Date|Event Date)[:\s]*([A-Za-z]+\s+\d{1,2},\s+\d{4}|\d{1,2}\/\d{1,2}\/\d{4})/i;
    const dateMatch = line.match(datePattern);
    if (dateMatch && !result.date) {
      result.date = dateMatch[1];
    }

    // Location pattern
    const locationPattern = /(?:Location|Venue|Place)[:\s]*(.+)/i;
    const locationMatch = line.match(locationPattern);
    if (locationMatch && !result.location) {
      result.location = locationMatch[1];
    }

    // Organizer pattern
    const organizerPattern = /(?:Organizer|Tournament Organizer|TO)[:\s]*(.+)/i;
    const organizerMatch = line.match(organizerPattern);
    if (organizerMatch && !result.organizer) {
      result.organizer = organizerMatch[1];
    }

    // Tournament type pattern
    const typePattern = /(?:Tournament Type|Format|Event Type)[:\s]*(.+)/i;
    const typeMatch = line.match(typePattern);
    if (typeMatch && !result.tournamentType) {
      result.tournamentType = typeMatch[1];
    }
  }

  // If we still don't have a name, use the first line that is not the sanction ID and not empty.
  if (!result.name) {
    for (const line of lines) {
      if (line && !line.includes(result.sanctionId || '')) {
        result.name = line;
        break;
      }
    }
  }

  return result;
}
```
