import { createClient } from '@/utils/supabase/server'
import { redirect } from 'next/navigation'
import { Role } from '@/lib/rbac'

export type TournamentAuthorizationResult = {
  isAuthorized: boolean
  tournament: any
  user: any
  profile: any
}

// Helper to detect Next.js redirect errors
export const isRedirectError = (error: unknown): error is { digest: string } => {
  return (
    error &&
    typeof error === 'object' &&
    'digest' in error &&
    typeof error.digest === 'string' &&
    error.digest.startsWith('NEXT_REDIRECT')
  )
}

// Unified error handler for auth errors
export const handleAuthError = (error: unknown): never => {
  if (isRedirectError(error)) {
    throw error // Let Next.js handle its own redirects
  }
  
  if (error instanceof Error && error.message === 'Tournament not found') {
    throw error // Let calling component handle notFound()
  }
  
  redirect('/?error=unauthorized')
}

/**
 * Get authenticated user with profile
 * @throws Redirects to login if not authenticated
 */
export async function getAuthenticatedUser() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()

  if (!user) {
    redirect('/login?redirect=/organizer')
  }

  const { data: profile, error } = await supabase
    .from('profiles')
    .select('role, pokemon_player_id, email')
    .eq('id', user.id)
    .single()

  if (error || !profile) {
    redirect('/login?message=Profile not found')
  }

  return { user, profile }
}

/**
 * Require organizer or admin role for route access
 * @throws Redirects if user lacks required role
 */
export async function requireOrganizerOrAdmin() {
  try {
    const { user, profile } = await getAuthenticatedUser()

    if (profile.role !== Role.ADMIN && profile.role !== Role.ORGANIZER) {
      redirect('/?error=unauthorized')
    }

    return { user, profile }
  } catch (error) {
    handleAuthError(error)
  }
}

/**
 * Authorize tournament management with three-tier access:
 * 1. User is tournament organizer (organizer_id)
 * 2. User is admin (role = 'admin')
 * 3. User has matching POP ID (pokemon_player_id = organizer_popid)
 */
export async function authorizeTournamentManagement(
  tournamentId: string
): Promise<TournamentAuthorizationResult> {
  try {
    const supabase = await createClient()
    
    // Get user and profile
    const { data: { user } } = await supabase.auth.getUser()
    if (!user) {
      redirect('/login')
    }

    const { data: profile, error: profileError } = await supabase
      .from('profiles')
      .select('role, pokemon_player_id')
      .eq('id', user.id)
      .single()

    if (profileError || !profile) {
      redirect('/login?message=Could not fetch profile')
    }

    // Get tournament
    const { data: tournament, error: tournamentError } = await supabase
      .from('tournaments')
      .select('*')
      .eq('id', tournamentId)
      .single()

    if (tournamentError || !tournament) {
      throw new Error('Tournament not found')
    }

    // Check authorization using three-tier access model
    let isAuthorized = tournament.organizer_id === user.id

    if (!isAuthorized && profile.role === Role.ADMIN) {
      isAuthorized = true
    }

    if (!isAuthorized && tournament.organizer_popid && profile.pokemon_player_id === tournament.organizer_popid) {
      isAuthorized = true
    }

    return {
      isAuthorized,
      tournament,
      user,
      profile
    }
  } catch (error) {
    handleAuthError(error)
  }
}

 * Check if user can create tournaments
 */
export async function canCreateTournaments(): Promise<boolean> {
  try {
    const supabase = await createClient()

      return false

      .eq('id', user.id)
      .single()

    return profile?.role === Role.ADMIN || profile?.role === Role.ORGANIZER
  } catch {
    return false
  }
}

### **2. `middleware.ts` - Optimized RBAC Middleware**

```typescript
import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'
import { cookies } from 'next/headers'
import { Role } from '@/lib/rbac'

// Cache profiles for 5 minutes to reduce DB queries
const profileCache = new Map<string, { role: Role; expires: number }>()

export async function middleware(request: NextRequest) {
  // Only protect /organizer/* routes
  if (!request.nextUrl.pathname.startsWith('/organizer')) {
    return NextResponse.next()
  }

  const cookieStore = await cookies()
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return cookieStore.getAll()
        },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            )
          } catch {
            // Ignore Server Component cookie setting
          }
        },
      },
    }
  )

  try {
    const { data: { session } } = await supabase.auth.getSession()

    if (!session) {
      const redirectUrl = new URL('/login', request.url)
      redirectUrl.searchParams.set('redirect', request.nextUrl.pathname)
      return NextResponse.redirect(redirectUrl)
    }

    // Check cache first
    const cachedProfile = profileCache.get(session.user.id)
    if (cachedProfile && cachedProfile.expires > Date.now()) {
      if (cachedProfile.role === Role.ADMIN || cachedProfile.role === Role.ORGANIZER) {
        return NextResponse.next()
      }
      return NextResponse.redirect(new URL('/', request.url))
    }

    // Query database if not cached
    const { data: profile } = await supabase
      .from('profiles')
      .select('role')
      .eq('id', session.user.id)
      .single()

    if (!profile || (profile.role !== Role.ADMIN && profile.role !== Role.ORGANIZER)) {
      return NextResponse.redirect(new URL('/', request.url))
    }

    // Update cache (5 minute TTL)
    profileCache.set(session.user.id, {
      role: profile.role,
      expires: Date.now() + 5 * 60 * 1000
    })

    return NextResponse.next()
  } catch (error) {
    console.error('Middleware authorization error:', error)
    return NextResponse.redirect(new URL('/login', request.url))
  }
}

export const config = {
  matcher: ['/organizer/:path*']
}

                <h3 className="font-semibold text-gray-700 mb-2">Tournament Details</h3>
                <p className="text-gray-600 whitespace-pre-line">{tournament.description}</p>
              </div>
            )}
          </div>
        </div>

        <style jsx global>{`
          @media print {
            .print-container {
              box-shadow: none;
              border: 1px solid #e5e7eb;
            }
            .no-print {
              display: none;
            }
          }
        `}</style>
      </div>
    )
  } catch (error) {
    if (error instanceof Error && error.message === 'Tournament not found') {
      notFound()
    }
    handleAuthError(error)
  }
}
```
