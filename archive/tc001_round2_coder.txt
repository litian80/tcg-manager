// scripts/setup-tc001.ts
import { createClient } from '@supabase/supabase-js';
import * as dotenv from 'dotenv';

dotenv.config();

const supabase = createClient(
  process.env.SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

async function setupTC001() {
  console.log('Setting up TC-001 prerequisites...');

    .from('tournaments')

      requires_deck_list: true,

      registrations_open: true,

      onConflict: 'name',
      ignoreDuplicates: true
    })
    .select()
    .single();

  // 2. Get test user data
  const { data: testUser } = await supabase.auth.admin.getUserByEmail(

  );
  
  // 3. Clean up existing registrations

    .from('tournament_players')
    .delete()

  await supabase
    .from('deck_lists')

    .eq('tournament_id', tournament.id)
    .eq('player_id', testUser.user?.id);

  console.log('Setup complete!');
  console.log(`Tournament URL: /tournament/${tournament.id}`);

setupTC001().catch(console.error);
