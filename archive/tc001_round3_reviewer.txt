    RETURNING pokemon_player_id INTO v_pokemon_player_id;

    RAISE NOTICE 'Test user pokemon_player_id: %', v_pokemon_player_id;
END $$;

-- Step 2: Create a tournament and clean up existing registration in one go

    FROM auth.users u

),
tournament_creation AS (
    INSERT INTO tournaments (
        id,
        name,
        organizer_id,
        organizer_popid,
        requires_deck_list,
        deck_list_submission_deadline,
        registrations_open,
        status,
        format,

        date,
        max_players,

        TRUE,

        'standard',

        NOW() + INTERVAL '72 hours',

        NOW(),
        NOW()
    FROM user_data
    RETURNING id, organizer_id
),
cleanup AS (
    DELETE FROM tournament_players tp
    USING user_data, tournament_creation tc

)
SELECT id as tournament_id FROM tournament_creation;
