  IF v_user_id IS NULL THEN
    RAISE EXCEPTION 'Test user % not found. Create user first.', test_user_email;
  END IF;
  
  -- Ensure profile exists with pokemon_player_id
  INSERT INTO profiles (id, pokemon_player_id, role, created_at, updated_at)
  VALUES (
    v_user_id,
    'TEST-' || v_user_id::TEXT, -- Generate a test pokemon_player_id
    'user',
    NOW(),
    NOW()
  )
  ON CONFLICT (id) 
  DO UPDATE SET 
    pokemon_player_id = COALESCE(profiles.pokemon_player_id, 'TEST-' || v_user_id::TEXT),
    updated_at = NOW()
  WHERE profiles.pokemon_player_id IS NULL;
  
  RAISE NOTICE 'Ensured test user has pokemon_player_id';
END $$;

    status,

    date,
    max_players,
    created_at,
    updated_at

    u.id,
    NULL,
    TRUE,

    TRUE,
    'pending',
    'standard',
    'Test Location',
    NOW() + INTERVAL '72 hours', -- Tournament in 3 days
    8,
    NOW(),
    NOW()
  FROM auth.users u
  WHERE u.email = 'test.user@example.com'
  LIMIT 1
  RETURNING id, name
),
-- 3. Clean up any existing registrations for test user
cleanup AS (
  DELETE FROM tournament_players tp
  WHERE tp.player_id IN (
    SELECT p.pokemon_player_id
    FROM profiles p
    WHERE p.id IN (

    if (!user.user) {
      throw new Error('Test user not found');

    const { error: profileError } = await supabaseAdmin

        id: user.user.id,

      }, { onConflict: 'id' });
    
    if (profileError) throw profileError;
    
    // 2. Create tournament
    const { data: tournament, error: tournamentError } = await supabaseAdmin
      .from('tournaments')
      .insert({
        name: `TC-001 Test Tournament - ${Date.now()}`,
        organizer_id: user.user.id,
        requires_deck_list: true,
        deck_list_submission_deadline: new Date(Date.now() + 48 * 60 * 60 * 1000).toISOString(),
        registrations_open: true,
        status: 'pending',
        format: 'standard',
        location: 'Test Location',
        date: new Date(Date.now() + 72 * 60 * 60 * 1000).toISOString(),
        max_players: 8
      })
      .select()
      .single();
    
    if (tournamentError) throw tournamentError;
    
    // 3. Clean existing registration
    const { error: cleanupError } = await supabaseAdmin
      .from('tournament_players')
      .delete()
      .eq('tournament_id', tournament.id)
      .eq('player_id', `TEST-${user.user.id.substring(0, 8)}`);
    
    if (cleanupError) throw cleanupError;
    
    console.log('✅ Setup complete!');
    console.log(`Tournament ID: ${tournament.id}`);
    console.log(`Test User ID: ${user.user.id}`);
    console.log(`Test User pokemon_player_id: TEST-${user.user.id.substring(0, 8)}`);
    console.log('\nVerification:');
    console.log(`1. Login as ${process.env.TEST_USER_EMAIL}`);
    console.log(`2. Navigate to /tournament/${tournament.id}`);
    console.log(`3. Verify: "Register" button visible, "No Deck Submitted" NOT visible`);
    
  } catch (error) {
    console.error('❌ Setup failed:', error);
    process.exit(1);
  }
}
