```typescript
// utils/deck-validator.ts - Improved version

export interface ParsedCard {
    qty: number;
    name: string;
    set: string;
    number: string;
    raw: string;
    isBasicEnergy?: boolean;
    category?: 'pokemon' | 'trainer' | 'energy';
    // For validation
    cardType?: string; // From DB: 'Pokémon', 'Trainer', 'Energy'
    supertype?: string;
    subtypes?: string[];
}

export function parseDeckList(deckText: string): DeckParseResult {
    const lines = deckText.trim().split('\n');
    const result: DeckParseResult = {
        Pokemon: [],
        Trainer: [],
        Energy: [],
        TotalCards: 0,
        Errors: []
    };
    
    // Normalize UTF-8 characters and handle multiple header formats
    const normalizedText = deckText
        .replace(/PokÃ©mon/g, 'Pokémon')
        .replace(/Pokemon/g, 'Pokémon')
        .replace(/Pok\u00e9mon/g, 'Pokémon');
    
    const normalizedLines = normalizedText.trim().split('\n');
    
    // More flexible category detection
    const categoryPatterns = {
        pokemon: /^(?:PokÃ©mon|Pokémon|Pokemon|PokÃ©mon:|Pokémon:|Pokemon:)/i,
        trainer: /^(?:Trainer|Trainer Cards|Trainer:|Trainers:)/i,
        energy: /^(?:Energy|Energies|Energy:|Energies:)/i
    };
    
    let currentCategory: keyof Omit<DeckParseResult, 'TotalCards' | 'Errors'> = 'Pokemon';
    
    for (let line of normalizedLines) {
        line = line.trim();
        if (!line) continue;
        
        // Detect category headers (case insensitive)
        if (categoryPatterns.pokemon.test(line)) {
            currentCategory = 'Pokemon';
            continue;
        } else if (categoryPatterns.trainer.test(line)) {
            currentCategory = 'Trainer';
            continue;
        } else if (categoryPatterns.energy.test(line)) {
            currentCategory = 'Energy';
            continue;
        }
        
        // Skip count lines
        if (/^(?:PokÃ©mon|Pokémon|Pokemon|Trainer|Energy):\s*\d+$/.test(line)) {
            continue;
        }
        
        // Enhanced patterns
        // Standard: "2 Entei V BRS 22"
        // Variants: "2 Entei V (BRS 22)", "2 Entei V - BRS 22", "2x Entei V BRS 22"
        const standardPattern = /^(\d+)[x\s]*(.+?)(?:\s+[-(]\s*)?([A-Z0-9-]{2,7})\s+(\d+)(?:\s*[)\]]?)?$/;
        
        // Basic Energy: "11 Basic {R} Energy SVE 2" or "11 Fire Energy" (no set)
        const basicEnergyPattern = /^(\d+)[x\s]*(Basic\s+{[^}]+}\s+Energy)(?:\s+([A-Z0-9-]{2,7})\s+(\d+))?$/i;
        
        // Try standard pattern first
        let match = line.match(standardPattern);
        if (match) {
            const [, qtyStr, name, setCode, cardNumber] = match;
            const qty = parseInt(qtyStr, 10);
            
            if (isNaN(qty) || qty <= 0 || qty > 99) {
                result.Errors.push({ line, message: `Invalid quantity: ${qtyStr}` });
                continue;
            }
            
            const cardObj: ParsedCard = {
                qty,
                name: name.trim(),
                set: setCode.trim().toUpperCase(),
                number: cardNumber.trim(),
                raw: line,
                category: currentCategory.toLowerCase() as ParsedCard['category']
            };
            
            result[currentCategory].push(cardObj);
            result.TotalCards += qty;
            continue;
        }
        
        // Try basic energy pattern
        match = line.match(basicEnergyPattern);
        if (match) {
            const [, qtyStr, name, setCode, cardNumber] = match;
            const qty = parseInt(qtyStr, 10);
            
            const cardObj: ParsedCard = {
                qty,
                name: name.trim(),
                set: (setCode || "ENERGY").trim().toUpperCase(),
                number: (cardNumber || "0").trim(),
                raw: line,
                isBasicEnergy: true,
                category: 'energy'
            };
            
            result.Energy.push(cardObj);
            result.TotalCards += qty;
            continue;
        }
        
        // If no match, check if it's a simple energy without "Basic"
        const simpleEnergyPattern = /^(\d+)[x\s]*([^{}\s]+(?:\s+{[^}]+})?\s+Energy)$/i;
        match = line.match(simpleEnergyPattern);
        if (match) {
            const [, qtyStr, name] = match;
            const qty = parseInt(qtyStr, 10);
            
            const cardObj: ParsedCard = {
                qty,
                name: name.trim(),
                set: "ENERGY",
                number: "0",
                raw: line,
                isBasicEnergy: true,
                category: 'energy'
            };
            
            result.Energy.push(cardObj);
            result.TotalCards += qty;
            continue;
        }
        
        result.Errors.push({
            line,
            message: "Unrecognized card format"
        });
    }
    
    return result;
}
```

// app/actions/deck-validation.ts - Enhanced version

export async function validateDeckListAction(
    deckText: string, 
    tournamentId?: string
): Promise<ValidationResult> {
    const supabase = await createClient();
    const result: ValidationResult = {
        isValid: true,
        errors: [],
        warnings: [],
        cardDetails: []
    };

    try {
        const parsed = parseDeckList(deckText);
        result.parsedDeck = parsed;

        // Early exit on parsing errors
        if (parsed.Errors.length > 0) {
            return {
                ...result,
                isValid: false,
                errors: parsed.Errors.map(e => `Line: "${e.line}" - ${e.message}`)
            };
        }

        // Standard deck size validation
        if (parsed.TotalCards !== 60) {
            result.errors.push(`Deck must contain exactly 60 cards, found ${parsed.TotalCards}`);
            result.isValid = false;
        }

        const allParsedCards = [...parsed.Pokemon, ...parsed.Trainer, ...parsed.Energy];
        if (allParsedCards.length === 0) {
            result.errors.push("No valid cards found in deck list");
            return { ...result, isValid: false };
        }

        // Hybrid Cache Lookup: Batch database queries
        const lookupPromises = [];
        
        // 1. Look up standard cards by set/number
        const standardCards = allParsedCards.filter(c => !c.isBasicEnergy);
        if (standardCards.length > 0) {
            // Group by set for more efficient querying
            const setGroups = new Map<string, string[]>();
            for (const card of standardCards) {
                const key = card.set;
                if (!setGroups.has(key)) setGroups.set(key, []);
                setGroups.get(key)!.push(card.number);
            }
            
            // Build optimized queries per set
            for (const [setCode, numbers] of setGroups) {
                const promise = supabase
                    .from('cards')
                    .select(`
                        id, 
                        name, 
                        card_number,
                        supertype,
                        subtypes,
                        card_type,
                        sets!inner(code, name),
                        equivalency_members!left(
                            equivalency_groups!inner(id, name)
                        )
                    `)
                    .eq('sets.code', setCode)
                    .in('card_number', numbers);
                
                lookupPromises.push(promise);
            }
        }
        
        // 2. Look up basic energy by name (if not found by set/number)
        const basicEnergyCards = parsed.Energy.filter(c => c.isBasicEnergy);
        if (basicEnergyCards.length > 0) {
            // Extract unique energy names
            const energyNames = [...new Set(basicEnergyCards.map(c => 
                c.name.replace(/\s*Energy$/i, '').trim()
            ))];
            
            const energyPromise = supabase
                .from('cards')
                .select(`
                    id, 
                    name, 
                    card_number,
                    supertype,
                    card_type,
                    sets!inner(code, name)
                `)
                .eq('supertype', 'Energy')
                .in('name', energyNames)
                .eq('sets.code', 'ENERGY'); // Basic energy set
                
            lookupPromises.push(energyPromise);
        }
        
        // Execute all lookups in parallel
        const lookupResults = await Promise.all(lookupPromises);
        
        // Process results
        const cardMap = new Map<string, any>();
        const nameToCardMap = new Map<string, any>();
        
        for (const { data: cards, error } of lookupResults) {
            if (error) {
                console.error("DB Lookup Error:", error);
                continue;
            }
            
            for (const card of cards || []) {
                const key = `${card.sets.code}-${card.card_number}`;
                cardMap.set(key, card);
                nameToCardMap.set(card.name.toLowerCase(), card);
            }
        }
        
        // Validate each parsed card
        const groupCounts = new Map<string, number>();
        const cardCounts = new Map<string, number>();
        const radiantCount = new Map<string, number>();
        const aceSpecCount = new Map<string, number>();
        const cardDetails = [];
        
        for (const parsedCard of allParsedCards) {
            let dbCard = null;
            
            // Find card in database
            if (parsedCard.isBasicEnergy) {
                // Try exact match first, then partial name match
                const energyName = parsedCard.name.replace(/\s*Energy$/i, '').trim();
                dbCard = nameToCardMap.get(energyName.toLowerCase()) || 
                        Array.from(nameToCardMap.values())
                            .find(c => c.name.toLowerCase().includes(energyName.toLowerCase()));
            } else {
                const key = `${parsedCard.set}-${parsedCard.number}`;
                dbCard = cardMap.get(key);
            }
            
            if (!dbCard) {
                result.errors.push(
                    `Card not found: ${parsedCard.name} (${parsedCard.set} ${parsedCard.number})`
                );
                result.isValid = false;
                continue;
            }
            
            // Track for 4-of rule
            const groupId = dbCard.equivalency_members?.[0]?.equivalency_groups?.id;
            const countKey = groupId ? `group:${groupId}` : `name:${dbCard.name.toLowerCase()}`;
            
            const currentGroupCount = groupCounts.get(countKey) || 0;
            groupCounts.set(countKey, currentGroupCount + parsedCard.qty);
            
            // Track specific card name for reporting
            const currentCardCount = cardCounts.get(dbCard.name) || 0;
            cardCounts.set(dbCard.name, currentCardCount + parsedCard.qty);
            
            // Special card type validations
            if (dbCard.subtypes?.includes('Radiant')) {
                const currentRadiant = radiantCount.get(dbCard.name) || 0;
                radiantCount.set(dbCard.name, currentRadiant + parsedCard.qty);
            }
            
            if (dbCard.subtypes?.includes('ACE SPEC')) {
                const currentAceSpec = aceSpecCount.get(dbCard.name) || 0;
                aceSpecCount.set(dbCard.name, currentAceSpec + parsedCard.qty);
            }
            
            // Store for later use in submission
            cardDetails.push({
                card_id: dbCard.id,
                quantity: parsedCard.qty,
                name: dbCard.name,
                set_code: dbCard.sets.code,
                card_number: dbCard.card_number,
                supertype: dbCard.supertype,
                subtypes: dbCard.subtypes
            });
        }
        
        result.cardDetails = cardDetails;
        
        // Enforce PTCG rules
        // 1. 4-of rule (standard cards)
        for (const [key, count] of groupCounts.entries()) {
            if (count > 4) {
                const isGroup = key.startsWith('group:');
                const identifier = isGroup 
                    ? "Equivalent cards group" 
                    : `Card "${key.split(':')[1]}"`;
                    
                result.errors.push(
                    `${identifier} exceeds 4-copy limit (found ${count})`
                );
                result.isValid = false;
            }
        }
        
        // 2. Radiant Pokemon rule (max 1 per deck total)
        const totalRadiant = Array.from(radiantCount.values()).reduce((a, b) => a + b, 0);
        if (totalRadiant > 1) {
            result.errors.push(`Maximum 1 Radiant Pokemon allowed, found ${totalRadiant}`);
            result.isValid = false;
        }
        
        // 3. ACE SPEC rule (max 1 per deck total)
        const totalAceSpec = Array.from(aceSpecCount.values()).reduce((a, b) => a + b, 0);
        if (totalAceSpec > 1) {
            result.errors.push(`Maximum 1 ACE SPEC card allowed, found ${totalAceSpec}`);
            result.isValid = false;
        }
        
        // 4. Basic Energy validation (no limits)
        // Note: Prism Star cards would have similar 1-of limit
        
        return result;

    } catch (e: any) {
        console.error("Validation Error:", e);
        return {
            isValid: false,
            errors: ["Unexpected validation error"],
            warnings: [],
            cardDetails: []
        };
    }
}

// app/actions/deck-submission.ts - Transaction-safe version

export async function submitDeckAction(
    tournamentId: string, 
    deckText: string
): Promise<ValidationResult> {
    const supabase = await createClient();
    
    // 1. Auth & Profile Check
    const { data: { user } } = await supabase.auth.getUser();
    if (!user) {
        return { isValid: false, errors: ["Authentication required"] };
    }

    const { data: profile } = await supabase
        .from('profiles')
        .select('pokemon_player_id')
        .eq('id', user.id)
        .single();
        
    if (!profile?.pokemon_player_id) {
        return { isValid: false, errors: ["Complete your profile with POP ID"] };
    }

    const playerId = profile.pokemon_player_id;

    // 2. Tournament validation
    const { data: tournament } = await supabase
        .from('tournaments')
        .select('requires_deck_list, deck_list_submission_deadline')
        .eq('id', tournamentId)
        .single();
        
    if (!tournament?.requires_deck_list) {
        return { isValid: false, errors: ["This tournament doesn't require deck lists"] };
    }
    
    if (tournament.deck_list_submission_deadline && 
        new Date() > new Date(tournament.deck_list_submission_deadline)) {
        return { isValid: false, errors: ["Deck submission deadline has passed"] };
    }

    // 3. Validate deck
    const validation = await validateDeckListAction(deckText, tournamentId);
    if (!validation.isValid || !validation.cardDetails) {
        return validation;
    }

    try {
        // Use a transaction via RPC to ensure atomicity
        const { data, error } = await supabase.rpc('submit_deck_transaction', {
            p_tournament_id: tournamentId,
            p_player_id: playerId,
            p_raw_text: deckText,
            p_card_data: validation.cardDetails.map(cd => ({
                card_id: cd.card_id,
                quantity: cd.quantity
            }))
        });

        if (error) {
            console.error("Transaction Error:", error);
            return { 
                isValid: false, 
                errors: ["Failed to save deck: " + error.message] 
            };
        }

        revalidatePath(`/tournament/${tournamentId}`);
        revalidatePath(`/tournament/${tournamentId}/deck`);
        
        return { ...validation, isValid: true };

    } catch (err: any) {
        console.error("Submission Error:", err);
        return { 
            isValid: false, 
            errors: ["Submission failed: " + err.message] 
        };
    }
}

-- Add this function to ensure atomic submission
CREATE OR REPLACE FUNCTION submit_deck_transaction(
    p_tournament_id UUID,
    p_player_id TEXT,
    p_raw_text TEXT,
    p_card_data JSONB
)
RETURNS UUID
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    v_deck_list_id UUID;
BEGIN
    -- Upsert deck list
    INSERT INTO deck_lists (
        tournament_id,
        player_id,
        raw_text,
        validation_status,
        submitted_at
    ) VALUES (
        p_tournament_id,
        p_player_id,
        p_raw_text,
        'valid',
        NOW()
    )
    ON CONFLICT (tournament_id, player_id) 
    DO UPDATE SET
        raw_text = EXCLUDED.raw_text,
        validation_status = EXCLUDED.validation_status,
        submitted_at = EXCLUDED.submitted_at
    RETURNING id INTO v_deck_list_id;
    
    -- Delete existing cards
    DELETE FROM deck_list_cards 
    WHERE deck_list_id = v_deck_list_id;
    
    -- Insert new cards from JSON array
    INSERT INTO deck_list_cards (deck_list_id, card_id, quantity, category)
    SELECT 
        v_deck_list_id,
        (elem->>'card_id')::UUID,
        (elem->>'quantity')::INTEGER,
        'main'
    FROM jsonb_array_elements(p_card_data) AS elem;
    
    RETURN v_deck_list_id;
EXCEPTION
    WHEN OTHERS THEN
        RAISE EXCEPTION 'Deck submission failed: %', SQLERRM;
END;
$$;
