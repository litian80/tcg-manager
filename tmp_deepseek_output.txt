```typescript
// utils/queue.ts

import { createClient } from '@/utils/supabase/server';
import { randomUUID } from 'crypto';
import { buildPaymentRedirectUrl } from '@/utils/payment';
import { tryDispatchNotification } from '@/utils/webhook-helpers';

export async function processQueue(tournamentId: string) {
  const supabase = await createClient();

  // Fetch tournament with queue settings and division capacities
  const { data: tournament, error: tError } = await supabase
    .from('tournaments')
    .select(`
      *,
      capacity_juniors,
      capacity_seniors,
      capacity_masters,
      enable_queue,
      queue_promotion_window_minutes,
      queue_batch_size,
      queue_paused,
      payment_required,
      payment_url,
      registration_open,
      registration_closes_at,
      registration_opens_at
    `)
    .eq('id', tournamentId)
    .single();

  if (tError || !tournament) {
    console.error('Tournament not found or error:', tError);
    return;
  }

  // Check if queue is enabled and not paused, and registration is open
  if (!tournament.enable_queue || tournament.queue_paused) {
    return;
  }

  // Check registration open and not closed
  if (!tournament.registration_open) {
    return;
  }

  if (tournament.registration_opens_at && new Date(tournament.registration_opens_at) > new Date()) {
    return;
  }

  if (tournament.registration_closes_at && new Date(tournament.registration_closes_at) < new Date()) {
    return;
  }

  // Define divisions
  const divisions = ['junior', 'senior', 'master'];

  for (const division of divisions) {
    const capacityColumn = `capacity_${division}s` as 'capacity_juniors' | 'capacity_seniors' | 'capacity_masters';
    const capacity = tournament[capacityColumn] || 0;

    // If capacity is 0, it's unlimited, so we can promote all queued players? But we have to be cautious.
    // We'll treat 0 as unlimited, so we can promote without limit.

    // Calculate current occupancy: registered, checked_in, and pending_payment (within promotion window)
    const promotionWindow = tournament.queue_promotion_window_minutes || 15;
    const { data: currentOccupants, error: ocError } = await supabase
      .from('tournament_players')
      .select('id')
      .eq('tournament_id', tournamentId)
      .eq('division', division)
      .in('registration_status', ['registered', 'checked_in', 'pending_payment'])
      .or(`and(registration_status.eq.pending_payment,queue_promoted_at.gte.${new Date(Date.now() - promotionWindow * 60000).toISOString()}),registration_status.in.(registered,checked_in)`);

    if (ocError) {
      console.error('Error fetching current occupants:', ocError);
      continue;
    }

    const currentCount = currentOccupants.length;
    const availableSlots = capacity === 0 ? 999999 : capacity - currentCount; // if unlimited, set a high number

    if (availableSlots <= 0) {
      continue;
    }

    // Get the next N queued players, where N = min(availableSlots, queue_batch_size)
    const batchSize = tournament.queue_batch_size || 5;
    const limit = Math.min(availableSlots, batchSize);

    const { data: queuedPlayers, error: qError } = await supabase
      .from('tournament_players')
      .select('*')
      .eq('tournament_id', tournamentId)
      .eq('division', division)
      .eq('registration_status', 'queued')
      .order('created_at', { ascending: true })
      .limit(limit);

    if (qError) {
      console.error('Error fetching queued players:', qError);
      continue;
    }

    // Promote each player
    for (const player of queuedPlayers) {
      let newStatus: string;
      let callbackToken: string | null = null;

      if (tournament.payment_required) {
        newStatus = 'pending_payment';
        callbackToken = randomUUID();
      } else {
        newStatus = 'registered';
      }

      const { error: updateError } = await supabase
        .from('tournament_players')
        .update({
          registration_status: newStatus,
          payment_callback_token: callbackToken,
          queue_promoted_at: new Date().toISOString(),
          ...(newStatus === 'pending_payment' ? { payment_pending_since: new Date().toISOString() } : {})
        })
        .eq('id', player.id);

      if (updateError) {
        console.error('Error promoting player:', updateError);
        continue;
      }

      // Send notification
      if (newStatus === 'pending_payment') {
        // Build payment URL and send to player (how? We can send an email or in-app notification? The player is waiting on the page, so we should update the UI in real-time.)
        // We can dispatch a webhook for the organiser, and the player's UI will update via real-time subscription.
        tryDispatchNotification(supabase, tournamentId, 'payment.pending', player.player_id, { division })
          .catch(() => {});
      } else {
        tryDispatchNotification(supabase, tournamentId, 'registration.confirmed', player.player_id, { division })
          .catch(() => {});
      }
    }
  }
}
```

```typescript
export async function POST(request: NextRequest) {
  try {
    const supabase = await createClient();

    } else {
      // Process all tournaments with queue enabled and registration open
      const { data: tournaments, error } = await supabase
        .from('tournaments')
        .select('id')
        .eq('enable_queue', true)
        .eq('queue_paused', false)
        .eq('registration_open', true)
        .lte('registration_opens_at', new Date().toISOString())
        .gte('registration_closes_at', new Date().toISOString());

      if (error) {

        return NextResponse.json({ error: 'Internal Server Error' }, { status: 500 });
      }

      for (const tournament of tournaments) {
        await processQueue(tournament.id);
      }
    }

    return NextResponse.json({ success: true });
  } catch (error) {
    console.error('Queue processing error:', error);
    return NextResponse.json({ error: 'Internal Server Error' }, { status: 500 });
  }
}
```
